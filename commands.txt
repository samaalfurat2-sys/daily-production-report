# commands.txt — Android OneDrive sign-in: keytool, adb, and log filtering
# ============================================================================
# Use these commands when setting up or debugging OneDrive authentication on
# Android.  See README.md → "Android OneDrive Sign-In Setup" for the full
# step-by-step guide.
# ============================================================================


# ── 1. Get your SHA-1 signing fingerprint ────────────────────────────────────

# Debug keystore (for development / emulator / CI):
keytool -list -v \
  -keystore ~/.android/debug.keystore \
  -alias androiddebugkey \
  -storepass android \
  -keypass android

# Release keystore (for Play Store / production APK):
# Replace the placeholders before running:
#   /path/to/your/release.jks  — path to your .jks / .keystore file
#   YOUR_KEY_ALIAS              — the alias used when the keystore was created
#   YOUR_STORE_PASSWORD         — the store password
keytool -list -v \
  -keystore /path/to/your/release.jks \
  -alias YOUR_KEY_ALIAS \
  -storepass YOUR_STORE_PASSWORD

# The output contains a line like:
#   SHA1: AA:BB:CC:DD:EE:FF:00:11:22:33:44:55:66:77:88:99:AA:BB:CC:DD
# Copy the SHA1 value and paste it into Azure Portal (Step 3 in README.md).


# ── 2. Build the debug APK ────────────────────────────────────────────────────

# From the Flutter project root (e.g. production_report_app/):
flutter build apk --debug

# Release APK (requires a keystore configured in android/app/build.gradle):
flutter build apk --release


# ── 3. Install the APK on a connected device / emulator ──────────────────────

adb install -r build/app/outputs/flutter-apk/app-debug.apk
# -r  : replace existing installation
# Use app-release.apk for a release build.


# ── 4. Launch the app via adb ─────────────────────────────────────────────────

# Replace "com.example.production_report_app" with your actual applicationId.
adb shell am start -n com.example.production_report_app/.MainActivity


# ── 5. Capture logcat for debugging ──────────────────────────────────────────

# Stream all logs filtered to your package (replace package name if needed):
adb logcat | grep -i "com.example.production_report_app"

# Broaden filter to auth / OAuth / MSAL related tags:
adb logcat | grep -iE "msal|oauth|auth|flutter|OneDriveAuth|graph"

# Filter only Flutter / Dart-level errors:
adb logcat -s flutter

# Show all logs with timestamps (useful for timing auth redirects):
adb logcat -v time | grep -iE "msal|oauth|OneDriveAuth"

# Save a snapshot of the log to a file for sharing with teammates:
adb logcat -d > /tmp/logcat_$(date +%Y%m%d_%H%M%S).txt

# Clear the log buffer before reproducing the issue (avoids noise):
adb logcat -c && adb logcat | grep -iE "msal|oauth|OneDriveAuth"


# ── 6. Verify the msauth:// intent-filter is registered ──────────────────────

# List all intent-filters for your package (look for "msauth" scheme):
adb shell dumpsys package com.example.production_report_app | grep -A5 "msauth"

# If the intent-filter is missing, check AndroidManifest.xml and rebuild.


# ── 7. Check device / emulator connectivity ──────────────────────────────────

# List connected devices:
adb devices

# Test HTTPS connectivity via the device shell.
# NOTE: curl may not be available on all Android devices/emulators.
# If curl is absent, use adb shell wget or test connectivity from within the app.
adb shell curl -I https://login.microsoftonline.com || \
  echo "curl not available — test network from within the app instead"


# ── NOTES ─────────────────────────────────────────────────────────────────────
# • Do NOT commit SHA-1 fingerprints or keystore passwords to source control.
# • The debug keystore is auto-generated by Android Studio at:
#     Windows: %USERPROFILE%\.android\debug.keystore
#     macOS/Linux: ~/.android/debug.keystore
# • For CI, generate a keystore and store it as a secret (not in the repo).
# • The device-code flow (default) does not require any of the above redirect
#   URI setup — it works on all platforms as long as INTERNET permission is set.
